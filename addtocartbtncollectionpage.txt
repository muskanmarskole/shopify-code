<script>
	{% for product in collection.products %}
	    {% for option in product.options %}
	      {% for variant in product.variants %}
	            $(".cus_v-{{ variant.id }}").each(function() {
	            $(this).on("click", function() {
	                            var dataSrc = $(this).attr("data-src");
	                            var varid= $(this).attr("variantID");
	                            var varianturl= $(this).attr("varianturl");
	                            var newurl=varianturl+varid
	                            $(".main-cus-img-{{ product.id }} .main-cus-img-c").attr("srcset", dataSrc);
	                            $(".main-cus-img-{{ product.id }} .main-cus-img-c").attr("vid",varid);
	                            $(".main-cus-img-{{ product.id }} .main-cus-img-c").attr("href",newurl);
	            });
});
	                                {% endfor %}
	                {% endfor %}
	{% endfor %}
</script>

        event.preventDefault(); // Prevent the default form submission

        var formData = new FormData(this);

        // Use Ajax to submit the form data without refreshing the page
        fetch('/cart/add', {
          method: 'POST',
          body: formData
        })
        .then(response => {
          if (!response.ok) {
            throw new Error('Network response was not ok');
          }

          // Check the content type of the response
          var contentType = response.headers.get('content-type');
          if (contentType && contentType.includes('application/json')) {
            return response.json(); // Parse JSON if content-type is JSON
          } else {
            return response.text(); // Return raw text if content-type is not JSON
          }
        })
        .then(data => {
          // Handle the response, e.g., update the cart count or show a success message

        })
        .catch(error => {
          console.error('Error adding product to cart:', error);
